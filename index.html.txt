<script>
/* =======================
   LANGUAGE TOGGLE
======================= */
let currentLang = 'pl';
const langToggle = document.getElementById('langToggle');

langToggle.addEventListener('click', () => {
  if (currentLang === 'pl') {
    document.body.classList.add('lang-english');
    currentLang = 'en';
  } else {
    document.body.classList.remove('lang-english');
    currentLang = 'pl';
  }
});

/* =======================
   NAVIGATION
======================= */
const navTabs = document.querySelectorAll('.nav-tab');
const pages = document.querySelectorAll('.page-content');

navTabs.forEach(tab => {
  tab.addEventListener('click', e => {
    e.preventDefault();
    const page = tab.dataset.page;

    navTabs.forEach(t => t.classList.remove('active'));
    pages.forEach(p => p.classList.remove('active'));

    tab.classList.add('active');
    const target = document.getElementById('page-' + page);
    if (target) target.classList.add('active');

    document.querySelector('.app-wrapper').scrollTop = 0;
  });
});

/* =======================
   LOGO ‚Üí HOME
======================= */
document.getElementById('logoContainer').addEventListener('click', () => {
  navTabs.forEach(t => t.classList.remove('active'));
  pages.forEach(p => p.classList.remove('active'));

  document.querySelector('.nav-tab[data-page="home"]').classList.add('active');
  document.getElementById('page-home').classList.add('active');

  document.querySelector('.app-wrapper').scrollTop = 0;
});

/* =======================
   FORM (FORMSPREE)

const form = document.getElementById('contactForm');
